   a) verify_github_signature(request, secret: str) -> bool:
      - Get X-Hub-Signature-256 header from request
      - If missing, return False
      - Compute expected = 'sha256=' + hmac.new(secret.encode(), request.body, hashlib.sha256).hexdigest()
      - Return hmac.compare_digest(expected, signature_header) - MUST use compare_digest for timing safety

   b) identify_service_from_webhook(payload: dict) -> Service | None:
      - Get workflow name from payload['workflow']['name'] or payload['workflow_run']['name']
      - Workflow name format: "CI - {workflow_name}" (from generate_manifest)
      - Parse to get workflow_name, then find CIWorkflow.objects.filter(name=workflow_name)
      - If found, return Service.objects.filter(ci_workflow=workflow).first()
      - Also try matching by repository URL: payload['repository']['html_url'] to service.repo_url
      - Return None if no match (log warning)

   c) @csrf_exempt build_webhook(request) view:
      - If not POST: return HttpResponse(status=200)
      - Parse JSON body
      - Get action = payload.get('action') - only process 'requested' (start) and 'completed'
      - Get repository URL from payload['repository']['html_url']
      - Find service via identify_service_from_webhook
      - If no service: log warning, return 200
      - Get project's default SCM connection
      - Get webhook_secret from connection config
      - If not verify_github_signature(request, webhook_secret): log warning, return 200
      - Get run_id = payload['workflow_run']['id']
      - Enqueue poll_build_details task with run_id, repo full_name, connection_id, service_id
      - Return HttpResponse(status=200)

      IMPORTANT: Always return 200 OK regardless of errors (security - prevents enumeration)

   def get_workflow_run(self, config: dict, repo_name: str, run_id: int) -> dict:
       """Fetch workflow run details from GitHub API."""
       g = self._get_github_client(config)
       repo = g.get_repo(repo_name)
       run = repo.get_workflow_run(run_id)

       return {
           'id': run.id,
           'run_number': run.run_number,
           'head_sha': run.head_sha,
           'head_branch': run.head_branch,
           'status': run.status,           # queued, in_progress, completed
           'conclusion': run.conclusion,   # success, failure, cancelled, etc.
           'created_at': run.created_at,
           'updated_at': run.updated_at,
           'html_url': run.html_url,
           'name': run.name,
           'event': run.event,
           'actor': {
               'login': run.actor.login if run.actor else None,
               'avatar_url': run.actor.avatar_url if run.actor else None,
           },
       }

2. Add poll_build_details task to core/tasks.py:

   @task(queue_name="build_updates")
   def poll_build_details(run_id: int, repo_name: str, connection_id: int, service_id: int) -> dict:
       """Poll GitHub API for workflow run details and update Build record."""
       from core.models import Build, IntegrationConnection, Service

       try:
           connection = IntegrationConnection.objects.get(id=connection_id)
       except IntegrationConnection.DoesNotExist:
           logger.error(f"Connection {connection_id} not found")
           return {"error": "Connection not found"}

       try:
           service = Service.objects.get(id=service_id)
       except Service.DoesNotExist:
           logger.error(f"Service {service_id} not found")
           return {"error": "Service not found"}

       plugin = connection.get_plugin()
       if not plugin:
           logger.error(f"Plugin not available for connection {connection.name}")
           return {"error": "Plugin not available"}

       # Fetch run details from GitHub
       try:
           run_data = plugin.get_workflow_run(connection.get_config(), repo_name, run_id)
       except Exception as e:
           logger.exception(f"Failed to fetch workflow run {run_id}")
           return {"error": str(e)}

       # Map GitHub status to our status
       status = Build.map_github_status(run_data['status'], run_data.get('conclusion'))

       # Calculate duration if completed
       duration = None
       completed_at = None
       if run_data['status'] == 'completed' and run_data['created_at'] and run_data['updated_at']:
           completed_at = run_data['updated_at']
           duration = int((completed_at - run_data['created_at']).total_seconds())

       # Update or create Build record
       build, created = Build.objects.update_or_create(
           github_run_id=run_id,
           defaults={
               'service': service,
               'run_number': run_data.get('run_number'),
               'status': status,
               'commit_sha': run_data.get('head_sha', ''),
               'branch': run_data.get('head_branch', ''),
               'author': run_data.get('actor', {}).get('login', ''),
               'author_avatar_url': run_data.get('actor', {}).get('avatar_url', ''),
               'ci_job_url': run_data.get('html_url', ''),
               'started_at': run_data.get('created_at'),
               'completed_at': completed_at,
               'duration_seconds': duration,
           }
       )

       # Activate service on first successful build
       if status == 'success':
           activate_service_on_first_success(build)

       logger.info(f"Build {build.id} {'created' if created else 'updated'}: {status}")
       return {"build_id": build.id, "status": status, "created": created}

3. Add activate_service_on_first_success helper function in core/tasks.py:

   def activate_service_on_first_success(build):
       """Transition service from draft to active on first successful build."""
       service = build.service
       if service.status != 'draft':
           return  # Already active or in error state

       # Check if this is actually the first success
       from core.models import Build
       has_prior_success = Build.objects.filter(
           service=service,
           status='success'
       ).exclude(id=build.id).exists()

       if has_prior_success:
           return  # Not the first success

       # Activate the service
       service.status = 'active'
       service.current_build_id = build.id
       service.save(update_fields=['status', 'current_build_id', 'updated_at'])
       logger.info(f"Service {service.name} activated on first successful build {build.id}")
