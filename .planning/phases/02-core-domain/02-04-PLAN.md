from core.models import Group, ProjectMembership

class AddMemberForm(forms.Form):
    """Form to add a group to a project with a role."""
    group = forms.ModelChoiceField(
        queryset=Group.objects.filter(status='active'),
        widget=forms.Select(attrs={
            'class': 'w-full px-3 py-2 bg-dark-bg border border-dark-border rounded-lg text-dark-text focus:outline-none focus:ring-2 focus:ring-dark-accent',
        })
    )
    project_role = forms.ChoiceField(
        choices=ProjectMembership.PROJECT_ROLES if hasattr(ProjectMembership, 'PROJECT_ROLES') else [
            ('owner', 'Owner'),
            ('contributor', 'Contributor'),
            ('viewer', 'Viewer'),
        ],
        widget=forms.Select(attrs={
            'class': 'w-full px-3 py-2 bg-dark-bg border border-dark-border rounded-lg text-dark-text focus:outline-none focus:ring-2 focus:ring-dark-accent',
        })
    )

    def __init__(self, *args, project=None, **kwargs):
        super().__init__(*args, **kwargs)
        if project:
            # Exclude groups already assigned to this project
            existing_group_ids = project.memberships.values_list('group_id', flat=True)
            self.fields['group'].queryset = Group.objects.filter(
                status='active'
            ).exclude(id__in=existing_group_ids)

from django.views.generic import View

from core.forms import AddMemberForm

class AddMemberModalView(LoginRequiredMixin, ProjectOwnerMixin, TemplateView):
    """Show modal to add a group to the project."""
    template_name = 'core/projects/add_member_modal.html'

    def get_context_data(self, **kwargs):

        context['form'] = AddMemberForm(project=self.project)

class AddMemberView(LoginRequiredMixin, ProjectOwnerMixin, View):
    """Handle adding a group to the project."""

    def post(self, request, project_uuid):
        form = AddMemberForm(request.POST, project=self.project)
        if form.is_valid():
            ProjectMembership.objects.create(
                project=self.project,
                group=form.cleaned_data['group'],
                project_role=form.cleaned_data['project_role'],
                added_by=request.user.username
            )
            # Return updated members tab
            memberships = self.project.memberships.select_related('group').order_by('project_role')
            return render(request, 'core/projects/_members_tab.html', {
                'project': self.project,
                'memberships': memberships,
                'user_project_role': self.user_project_role,
            })
        # Re-render modal with errors
        return render(request, 'core/projects/add_member_modal.html', {
            'project': self.project,
            'form': form,
        })


class RemoveMemberView(LoginRequiredMixin, ProjectOwnerMixin, View):
    """Remove a group from the project."""

    def delete(self, request, project_uuid, group_uuid):
        from core.models import Group
        group = get_object_or_404(Group, uuid=group_uuid)
        ProjectMembership.objects.filter(
            project=self.project,
            group=group
        ).delete()
        # Return updated members tab
        memberships = self.project.memberships.select_related('group').order_by('project_role')
        return render(request, 'core/projects/_members_tab.html', {
            'project': self.project,
            'memberships': memberships,
            'user_project_role': self.user_project_role,
        })

from django.views.generic import CreateView, UpdateView, DeleteView, DetailView
from django.contrib.auth.mixins import LoginRequiredMixin
from django.shortcuts import get_object_or_404, redirect
from django.urls import reverse
from django.db import transaction
from django.contrib import messages

from core.models import Project, Environment
from core.forms import EnvironmentCreateForm, EnvironmentEditForm
from core.permissions import ProjectOwnerMixin


class EnvironmentCreateView(LoginRequiredMixin, ProjectOwnerMixin, CreateView):
    model = Environment
    form_class = EnvironmentCreateForm
    template_name = 'core/environments/create.html'

    def form_valid(self, form):
        with transaction.atomic():
            form.instance.project = self.project

            # Check if this is the first environment
            is_first = not Environment.objects.filter(project=self.project).exists()
            if is_first:
                form.instance.is_default = True

            self.object = form.save()

        messages.success(self.request, f'Environment "{self.object.name}" created.')
        return redirect('projects:detail', project_uuid=self.project.uuid)

    def get_context_data(self, **kwargs):
        context = super().get_context_data(**kwargs)
        context['project'] = self.project
        return context

    def get_success_url(self):
        return reverse('projects:detail', kwargs={'project_uuid': self.project.uuid}) + '?tab=environments'


class EnvironmentDetailView(LoginRequiredMixin, ProjectOwnerMixin, UpdateView):
    """Environment detail/edit view."""
    model = Environment
    form_class = EnvironmentEditForm
    template_name = 'core/environments/detail.html'
    context_object_name = 'environment'

    def get_object(self):
        return get_object_or_404(
            Environment,
            uuid=self.kwargs['env_uuid'],
            project=self.project
        )

    def form_valid(self, form):
        with transaction.atomic():
            env = form.save(commit=False)

            # Handle default change
            if env.is_default:
                # Unmark other defaults
                Environment.objects.filter(
                    project=self.project,
                    is_default=True
                ).exclude(pk=env.pk).update(is_default=False)

            env.save()

        messages.success(self.request, f'Environment "{env.name}" updated.')
        return redirect('projects:environment_detail', project_uuid=self.project.uuid, env_uuid=env.uuid)

    def get_context_data(self, **kwargs):
        context = super().get_context_data(**kwargs)
        context['project'] = self.project
        # Get merged env vars for display
        context['merged_env_vars'] = get_merged_env_vars(self.project, self.object)
        return context


class EnvironmentDeleteView(LoginRequiredMixin, ProjectOwnerMixin, DeleteView):
    model = Environment
    template_name = 'core/environments/confirm_delete.html'

    def get_object(self):
        return get_object_or_404(
            Environment,
            uuid=self.kwargs['env_uuid'],
            project=self.project
        )

    def get_success_url(self):
        return reverse('projects:detail', kwargs={'project_uuid': self.project.uuid}) + '?tab=environments'


def get_merged_env_vars(project, environment=None):
    """
    Merge env_vars: Project -> Environment
    Locked vars cannot be overridden downstream.
    Returns: list of {"key": str, "value": str, "lock": bool, "source": str, "inherited": bool}
    """
    merged = {}

    # Start with project vars
    for var in (project.env_vars or []):
        merged[var['key']] = {
            'value': var['value'],
            'lock': var.get('lock', False),
            'source': 'project',
            'inherited': False,
        }

    if environment:
        for var in (environment.env_vars or []):
            key = var['key']
            if key in merged and merged[key]['lock']:
                # Locked at project level - keep project value, mark as inherited
                merged[key]['inherited'] = True
            else:
                # Can override
                if key in merged:
                    # Override existing
                    merged[key] = {
                        'value': var['value'],
                        'lock': var.get('lock', False),
                        'source': 'environment',
                        'inherited': False,
                    }
                else:
                    # New env var
                    merged[key] = {
                        'value': var['value'],
                        'lock': var.get('lock', False),
                        'source': 'environment',
                        'inherited': False,
                    }

    # Convert to list format
    return [
        {'key': k, **v}
        for k, v in sorted(merged.items())
    ]
