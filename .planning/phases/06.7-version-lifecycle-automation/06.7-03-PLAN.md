**Cleanup function and scheduled task (core/tasks.py):**

Create a `cleanup_old_versions()` function (not a task, a plain function) that can be called both from the cron task and the management command:

```python
def cleanup_old_versions() -> dict:
    """Clean up old version manifest content and delete unreferenced versions."""
    from django.utils import timezone
    from datetime import timedelta

    config = SiteConfiguration.get_instance()
    retention_days = config.version_retention_days
    cutoff = timezone.now() - timedelta(days=retention_days)

    # Step 1: Clear manifest_content for old authorized/revoked versions
    # Keep manifest_hash for verification. Only clear content, not the record.
    old_versions = CIWorkflowVersion.objects.filter(
        status__in=["authorized", "revoked"],
        published_at__lt=cutoff,
        manifest_content__isnull=False,
    ).exclude(manifest_content="")
    content_cleared = old_versions.update(manifest_content="")

    # Step 2: Delete version records that meet ALL criteria:
    # - No Build references (Build.workflow_version FK)
    # - Not the latest authorized version for their workflow
    # - Older than retention period
    # - Status is "revoked" (don't delete authorized versions that are still latest)
    deleted_count = 0
    for version in CIWorkflowVersion.objects.filter(
        status="revoked",
        published_at__lt=cutoff,
    ):
        # Check no builds reference this version
        if version.builds.exists():
            continue
        # Check not pinned by any service
        if version.pinned_services.exists():
            continue
        version.delete()
        deleted_count += 1

    # Update last_cleanup_at
    config.last_cleanup_at = timezone.now()
    config.save(update_fields=["last_cleanup_at"])

    return {
        "content_cleared": content_cleared,
        "versions_deleted": deleted_count,
    }
```

```python
class CIConfigSettingsView(LoginRequiredMixin, AdminRequiredMixin, View):
    template_name = "core/settings/ci_config.html"

    def get(self, request):
        config = SiteConfiguration.get_instance()
        form = RetentionSettingsForm(instance=config)
        return render(request, self.template_name, {
            "active_section": "settings",
            "active_settings_section": "ci_config",
            "form": form,
            "config": config,
        })

    def post(self, request):
        config = SiteConfiguration.get_instance()
        # Check if this is a cleanup request
        if "run_cleanup" in request.POST:
            from core.tasks import cleanup_old_versions
            result = cleanup_old_versions()
            msg = (
                f"Cleanup complete: {result['content_cleared']} manifest(s) cleared, "
                f"{result['versions_deleted']} version(s) deleted."
            )
            messages.success(request, msg)
            # Refresh config to get updated last_cleanup_at
            config.refresh_from_db()
            form = RetentionSettingsForm(instance=config)
        else:
            form = RetentionSettingsForm(request.POST, instance=config)
            if form.is_valid():
                form.save()
                messages.success(request, "Retention settings saved.")
            # else form will show errors
        return render(request, self.template_name, {
            "active_section": "settings",
            "active_settings_section": "ci_config",
            "form": form,
            "config": config,
        })
```

Note: The "Run Cleanup Now" button runs cleanup synchronously in the view for simplicity (cleanup is fast -- just DB queries). If needed, it could enqueue to background worker, but the research noted the settings page should "enqueue the task to background worker." Per Claude's discretion, run synchronously since it is just DB operations (no external API calls). If the operation becomes slow, it can be moved to background later.
