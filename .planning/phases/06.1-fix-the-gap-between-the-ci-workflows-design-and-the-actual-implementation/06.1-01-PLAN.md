```python
def extract_manifest_id(self, run_data: dict) -> str | None:
    name = run_data.get("name", "")
    if name.startswith("ci-"):
        manifest_id = f".github/workflows/{name}.yml"
        if self.MANIFEST_ID_PATTERN.match(manifest_id):
            return manifest_id
    return None
```

5. `fetch_manifest_content(self, config, repo_name, manifest_id, commit_sha)`:
```python
def fetch_manifest_content(
    self, config: dict, repo_name: str, manifest_id: str, commit_sha: str
) -> str | None:
    try:
        g = self._get_github_client(config)
        repo = g.get_repo(repo_name)
        content_file = repo.get_contents(manifest_id, ref=commit_sha)
        if isinstance(content_file, list):
            return None  # directory, not file
        return content_file.decoded_content.decode("utf-8")
    except Exception:
        return None
```
