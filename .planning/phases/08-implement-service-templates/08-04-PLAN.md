**theme/templates/base.html** — Register `templateSelector` Alpine.data component alongside existing `ciWorkflowSelector`. Place it right after the ciWorkflowSelector registration. Follow the exact same pattern (function syntax, not arrow functions, single expressions per handler):

```javascript
Alpine.data('templateSelector', function(initialTpl, initialVer, versionsMap) {
    return {
        tplVal: String(initialTpl),
        tplOpen: false,
        verVal: String(initialVer),
        verOpen: false,
        versions: [],
        init: function() {
            this.versions = versionsMap[this.tplVal] || [];
        },
        pickTpl: function(id) {
            this.tplVal = id;
            this.tplOpen = false;
            this.versions = versionsMap[id] || [];
            this.verVal = this.versions.length > 0 ? this.versions[0].tag : '';
            this.verOpen = false;
        },
        pickVer: function(tag) {
            this.verVal = tag;
            this.verOpen = false;
        }
    };
});
```
