  2. In `plugins/github/plugin.py`, implement `provision_ci_variables()` on `GitHubPlugin`:
     - Use PyGithub's repository variables API: `repo.create_variable(name, value)` for creation.
     - Implement the try/create-catch-409/update pattern since GitHub's variable creation is not idempotent:
       ```python
       def provision_ci_variables(self, config: dict, repo_name: str, variables: dict[str, str]) -> dict:
           """Provision GitHub Actions repository variables."""
           g = self._get_github_client(config)
           repo = g.get_repo(repo_name)
           results = {}
           for name, value in variables.items():
               try:
                   repo.create_variable(name, value)
                   results[name] = "created"
               except GithubException as e:
                   if e.status == 409:
                       # Variable already exists -- update it
                       try:
                           var = repo.get_variable(name)
                           var.edit(name, value)
                           results[name] = "updated"
                       except Exception as update_err:
                           results[name] = f"error: {update_err}"
                   else:
                       results[name] = f"error: {e.data.get('message', str(e)) if isinstance(e.data, dict) else str(e)}"
           return results
       ```

       # Provision CI variables (non-blocking)
       if service.repo_url and connection:
           from plugins.base import CICapableMixin, get_ci_plugin_for_engine
           plugin = connection.get_plugin()
           if isinstance(plugin, CICapableMixin):
               try:
                   from core.git_utils import parse_git_url
                   parsed = parse_git_url(service.repo_url)
                   if parsed:
