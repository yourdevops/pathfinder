**core/validators.py:**
```python
"""Validation utilities for DNS-compatible naming."""
import re
from django.core.validators import RegexValidator

# RFC 1123 label: lowercase letters, numbers, hyphens
# Must start with letter, max 63 chars, no leading/trailing hyphens
# Single letter is also valid
DNS_LABEL_REGEX = r'^[a-z][a-z0-9-]{0,61}[a-z0-9]$|^[a-z]$'

dns_label_validator = RegexValidator(
    regex=DNS_LABEL_REGEX,
    message='Name must be DNS-compatible: lowercase letters, numbers, and hyphens only. '
            'Must start with a letter, max 63 characters.',
    code='invalid_dns_label'
)
```

**core/converters.py:**
```python
"""Custom URL path converters."""

class DnsLabelConverter:
    """
    URL path converter for DNS-compatible names (RFC 1123 label format).

    Matches: lowercase letters, numbers, hyphens
    Must start with a letter, max 63 characters
    """
    regex = r'[a-z][a-z0-9-]{0,61}[a-z0-9]|[a-z]'

    def to_python(self, value):
        return value

    def to_url(self, value):
        return value
```
