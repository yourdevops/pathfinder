**Register `envVarWizard` Alpine component:**

In `step_configuration.html`, add the Alpine component in `{% block extra_head %}`:

```html
{% block extra_head %}
<script>
document.addEventListener('alpine:init', function() {
    Alpine.data('envVarWizard', function() {
        return {
            vars: [],
            ptfVars: [],
            showDesc: {},

            initFromField: function(fieldId, ptfData) {
                this.ptfVars = ptfData;
                var el = document.getElementById(fieldId);
                if (el && el.value) {
                    try { this.vars = JSON.parse(el.value); } catch(e) { this.vars = []; }
                }
            },

            addVar: function() {
                this.vars.push({ key: '', value: '', lock: false, description: '' });
                this.syncField();
            },

            removeVar: function(idx) {
                this.vars.splice(idx, 1);
                this.syncField();
            },

            updateKey: function(idx, val) {
                this.vars[idx].key = val.toUpperCase().replace(/[^A-Z0-9_]/g, '_');
                this.syncField();
            },

            updateValue: function(idx, val) {
                this.vars[idx].value = val;
                this.syncField();
            },

            updateDesc: function(idx, val) {
                this.vars[idx].description = val;
                this.syncField();
            },

            toggleLock: function(idx) {
                this.vars[idx].lock = !this.vars[idx].lock;
                this.syncField();
            },

            toggleDesc: function(key) {
                this.showDesc[key] = !this.showDesc[key];
            },

            syncField: function() {
                var el = document.getElementById('id_3-env_vars_json');
                if (el) { el.value = JSON.stringify(this.vars); }
            }
        };
    });
});
</script>
{% endblock %}
```

**Important CSP notes:** All methods use `function()` syntax (not arrow functions). No semicolons in Alpine directives. Directives call single methods like `@click="addVar()"`, `@input="updateKey(index, $event.target.value)"`.

</div>
