**Add _validate_output_references() function:**

Add a new function after `_validate_step_order()`:

```python
def _validate_output_references(steps_data, ci_plugin=None):
    """Validate all output references in step input configs.

    For each step's input_config values, detect output references
    and validate:
    1. Referenced step exists in the workflow
    2. Referenced step appears before the consuming step
    3. Referenced output exists in the step's outputs_schema
    """
    if not ci_plugin:
        return []

    errors = []
    # Build slug -> (index, CIStep) mapping from ordered steps
    step_map = {}  # slug -> (index, CIStep)
    ordered_steps = []

    for i, entry in enumerate(steps_data):
        try:
            ci_step = CIStep.objects.get(uuid=entry.get("id"))
            step_map[ci_step.slug] = (i, ci_step)
            ordered_steps.append((i, ci_step, entry.get("input_config", {})))
        except CIStep.DoesNotExist:
            continue

    for step_index, ci_step, input_config in ordered_steps:
        for input_name, value in input_config.items():
            if not isinstance(value, str):
                continue
            ref = ci_plugin.parse_output_reference(value)
            if not ref:
                continue
            ref_slug = ref["step_slug"]
            ref_output = ref["output_name"]

            if ref_slug not in step_map:
                errors.append(
                    f'"{ci_step.name}" input "{input_name}": '
                    f'step "{ref_slug}" not found in workflow'
                )
                continue
            ref_index, ref_step = step_map[ref_slug]
            if ref_index >= step_index:
                errors.append(
                    f'"{ci_step.name}" input "{input_name}": '
                    f'step "{ref_slug}" must appear before this step'
                )
                continue
            outputs = ref_step.outputs_schema or {}
            if ref_output not in outputs:
                errors.append(
                    f'"{ci_step.name}" input "{input_name}": '
                    f'output "{ref_output}" not found on step "{ref_slug}"'
                )
    return errors
```

**4. Add helper methods to the workflowComposer Alpine component:**

```javascript
setConfigTab: function(index, tab) {
    this.steps[index].configTab = tab;
},

hasOutputs: function(step) {
    return step.outputs_schema && Object.keys(step.outputs_schema).length > 0;
},

outputEntries: function(step) {
    var entries = [];
    var schema = step.outputs_schema || {};
    for (var key in schema) {
        if (schema.hasOwnProperty(key)) {
            entries.push({name: key, def: schema[key]});
        }
    }
    return entries;
},

_parseOutputRef: function(value) {
    var m = value.trim().match(
        /^\$\{\{\s*steps\.([a-z0-9][a-z0-9-]*)\.outputs\.([a-z0-9][a-z0-9_-]*)\s*\}\}$/
    );
    if (m) return {stepSlug: m[1], outputName: m[2]};
    return null;
},

validateInputValue: function(stepIndex, inputName, value) {
    var ref = this._parseOutputRef(value);
    if (!ref) {
        this.steps[stepIndex].validation_errors[inputName] = null;
        return;
    }
    // Find referenced step by slug
    var refStepIndex = -1;
    var refStep = null;
    for (var i = 0; i < this.steps.length; i++) {
        if (this.steps[i].slug === ref.stepSlug) {
            refStepIndex = i;
            refStep = this.steps[i];
            break;
        }
    }
    if (!refStep) {
        this.steps[stepIndex].validation_errors[inputName] = "Step '" + ref.stepSlug + "' not found in this workflow";
        return;
    }
    if (refStepIndex >= stepIndex) {
        this.steps[stepIndex].validation_errors[inputName] = "Step '" + ref.stepSlug + "' must appear before this step";
        return;
    }
    var outputs = refStep.outputs_schema || {};
    if (!outputs[ref.outputName]) {
        this.steps[stepIndex].validation_errors[inputName] = "Output '" + ref.outputName + "' not found on step '" + ref.stepSlug + "'";
        return;
    }
    this.steps[stepIndex].validation_errors[inputName] = null;
},

revalidateAllOutputRefs: function() {
    for (var i = 0; i < this.steps.length; i++) {
        var config = this.steps[i].input_config || {};
        for (var key in config) {
            if (config.hasOwnProperty(key) && typeof config[key] === 'string') {
                this.validateInputValue(i, key, config[key]);
            }
        }
    }
}
```
