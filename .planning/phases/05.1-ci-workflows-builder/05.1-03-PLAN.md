    **Alpine.js data structure for composer:**
    ```javascript
    {
      steps: [],
      addStep(step) {
        // step = {id, name, phase, description, inputs_schema}
        this.steps.push({...step, order: this.steps.length, input_config: {}});
        this.updateOrders();
      },
      removeStep(index) {
        this.steps.splice(index, 1);
        this.updateOrders();
      },
      moveUp(index) {
        if (index > 0) {
          [this.steps[index-1], this.steps[index]] = [this.steps[index], this.steps[index-1]];
          this.updateOrders();
        }
      },
      moveDown(index) {
        if (index < this.steps.length - 1) {
          [this.steps[index], this.steps[index+1]] = [this.steps[index+1], this.steps[index]];
          this.updateOrders();
        }
      },
      updateOrders() {
        this.steps.forEach((s, i) => s.order = i);
      },
      updateStepConfig(index, inputName, value) {
        if (!this.steps[index].input_config) this.steps[index].input_config = {};
        this.steps[index].input_config[inputName] = value;
      }
    }
    ```
