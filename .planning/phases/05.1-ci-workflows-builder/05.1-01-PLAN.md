       - `name = CharField(max_length=63, unique=True, validators=[dns_label_validator])`

       - `default_branch = CharField(max_length=100, default='main')`

       - `scan_error = TextField(blank=True)`

       - `created_by = CharField(max_length=150, blank=True)`
       - `created_at = DateTimeField(auto_now_add=True)`
       - `updated_at = DateTimeField(auto_now=True)`
       - `Meta: db_table = 'core_steps_repository'`
       - `__str__` returns `self.name`

    2. `RuntimeFamily` model:
       - `id = BigAutoField(primary_key=True)`
       - `repository = ForeignKey(StepsRepository, on_delete=CASCADE, related_name='runtimes')`
       - `name = CharField(max_length=63)` (e.g., 'python', 'node')

       - `Meta: db_table = 'core_runtime_family', unique_together = [['repository', 'name']]`
       - `__str__` returns `f"{self.name} ({self.repository.name})"`

    3. `CIStep` model:
       - `id = BigAutoField(primary_key=True)`
       - `uuid = UUIDField(default=uuid.uuid4, editable=False, unique=True, db_index=True)`
       - `repository = ForeignKey(StepsRepository, on_delete=CASCADE, related_name='steps')`
       - `directory_name = CharField(max_length=255)` (e.g., 'setup-python')
       - `name = CharField(max_length=255)` (from action.yml 'name')
       - `description = TextField(blank=True)` (from action.yml 'description')
       - `phase = CharField(max_length=20, choices=[('setup','Setup'),('build','Build'),('test','Test'),('package','Package')], blank=True)`
       - `runtime_constraints = JSONField(default=dict)` (e.g., {"python": ">=3.10"})
       - `tags = JSONField(default=list)`
       - `produces = JSONField(null=True, blank=True)` (e.g., {"type": "container-image"})
       - `inputs_schema = JSONField(default=dict)` (full inputs from action.yml)
       - `commit_sha = CharField(max_length=40, blank=True)`
       - `raw_metadata = JSONField(default=dict)` (full parsed action.yml)
       - `created_at = DateTimeField(auto_now_add=True)`
       - `updated_at = DateTimeField(auto_now=True)`
       - `Meta: db_table = 'core_ci_step', unique_together = [['repository', 'directory_name']]`
       - `__str__` returns `self.name`

    4. `CIWorkflow` model:
       - `id = BigAutoField(primary_key=True)`
       - `uuid = UUIDField(default=uuid.uuid4, editable=False, unique=True, db_index=True)`
       - `name = CharField(max_length=63, unique=True, validators=[dns_label_validator])`
       - `description = TextField(blank=True)`
       - `runtime_family = CharField(max_length=63)` (e.g., 'python')
       - `runtime_version = CharField(max_length=20)` (e.g., '3.12')
       - `artifact_type = CharField(max_length=50, blank=True)` (derived from last package step)
       - `created_by = CharField(max_length=150, blank=True)`
       - `created_at = DateTimeField(auto_now_add=True)`
       - `updated_at = DateTimeField(auto_now=True)`
       - `Meta: db_table = 'core_ci_workflow'`
       - `__str__` returns `self.name`

    5. `CIWorkflowStep` model:
       - `id = BigAutoField(primary_key=True)`
       - `workflow = ForeignKey(CIWorkflow, on_delete=CASCADE, related_name='workflow_steps')`
       - `step = ForeignKey(CIStep, on_delete=PROTECT, related_name='workflow_usages')`
       - `order = IntegerField()`
       - `input_config = JSONField(default=dict)` (per-step overrides)
       - `created_at = DateTimeField(auto_now_add=True)`
       - `Meta: db_table = 'core_ci_workflow_step', unique_together = [['workflow', 'order']], ordering = ['order']`
       - `__str__` returns `f"{self.workflow.name} - Step {self.order}: {self.step.name}"`
